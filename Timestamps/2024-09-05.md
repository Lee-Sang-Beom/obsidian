#### 네이버 지도 바운더리 밖에있는 요소는 안보이게하기
---
- 네이버 지도 API에서는 지도 바운더리 밖에 있는 마커를 자동으로 필터링하는 기능은 제공되지 않습니다. 그러나 이를 직접 구현할 수 있습니다. 지도 바운더리를 구하고 마커의 좌표가 그 안에 있을 때만 마커를 표시하면 됩니다.

>1. 지도 바운더리 구하기
- 네이버 지도 API에서 현재 지도 화면의 바운더리를 구할 수 있는 방법은 `getBounds()` 메서드를 사용하는 것입니다. 이 메서드는 현재 지도 화면의 좌상단과 우하단 좌표를 반환합니다.
```js
var map = new naver.maps.Map(...);  // 네이버 지도 객체 생성

// 현재 지도 바운더리 얻기
var bounds = map.getBounds();

// 바운더리 좌상단 (위쪽, 왼쪽) 좌표
var northEast = bounds.getNE();

// 바운더리 우하단 (아래쪽, 오른쪽) 좌표
var southWest = bounds.getSW();
```

> 2. 마커 필터링
- 마커가 바운더리 안에 있는지 확인하는 로직은 아래와 같이 구현할 수 있습니다. 마커의 위도와 경도가 바운더리 안에 있을 때만 마커를 지도에 표시합니다.
```jsx
var markers = [ /* 마커의 좌표 리스트 */ ];

// 마커가 바운더리 안에 있는지 확인하는 함수
function isMarkerInBounds(markerPosition, bounds) {
    var lat = markerPosition.lat,
        lng = markerPosition.lng;

    var northEast = bounds.getNE();
    var southWest = bounds.getSW();

    return lat <= northEast.lat && lat >= southWest.lat && lng <= northEast.lng && lng >= southWest.lng;
}

// 마커 필터링 후 지도에 추가
markers.forEach(function(markerPosition) {
    if (isMarkerInBounds(markerPosition, map.getBounds())) {
        // 바운더리 안에 있는 마커만 지도에 표시
        var marker = new naver.maps.Marker({
            position: new naver.maps.LatLng(markerPosition.lat, markerPosition.lng),
            map: map
        });
    }
});
```


> 3. 지도 이동시 마커 갱신
- 지도가 이동되거나 확대/축소될 때도 바운더리를 기준으로 마커를 필터링하려면 `idle` 이벤트를 사용하여 마커를 갱신할 수 있습니다.
```js

        } else {
            // 마커가 바운더리 밖에 있으면 삭제하거나 숨김

    });
});
```
