# Nginx를 실행할 사용자 계정
user  nginx;

# 동시에 일할 프로세스(직원) 수 - 1명의 워커가 일함
worker_processes  1;

# 에러 발생시 기록할 파일 위치 (warn 레벨 이상만 기록)
error_log  /var/log/nginx/error.log warn;

# Nginx 프로세스 ID를 저장할 파일 위치 (프로그램 관리용)
pid        /var/run/nginx.pid;

# 이벤트 처리 설정
events {
  # 워커 1명이 동시에 처리할 수 있는 최대 연결 수 (1024명까지 동시 접속 가능)
  worker_connections  1024;
}

# HTTP 서버 설정 시작
http {
  # 파일 확장자별 MIME 타입 정의 파일 불러오기 (.jpg는 이미지, .html은 HTML 등)
  include       /etc/nginx/mime.types;
  
  # MIME 타입을 모르는 파일의 기본 타입 (다운로드 파일로 처리)
  default_type  application/octet-stream;
  
  # 추가 MIME 타입 정의 - wav 파일은 오디오로 인식
  types {
    audio/wav wav;
  }
  
  # 접속 로그 형식 정의 - 누가, 언제, 무엇을, 어떤 결과로 요청했는지 기록
  log_format  main  '$remote_addr - $remote_user [$time_local] "$request" '
                    '$status $body_bytes_sent "$http_referer" '
                    '"$http_user_agent" "$http_x_forwarded_for"';
  
  # 접속 로그를 저장할 파일 위치 (위에서 정의한 main 형식으로 기록)
  access_log  /var/log/nginx/access.log  main;
  
  # 에러 로그 저장 위치
  error_log   /var/log/nginx/error.log;
  
  # 파일 전송 최적화 옵션 켜기 (전송 속도 향상)
  sendfile        on;
  
  # 연결 유지 시간 - 65초 동안 연결 유지 (매번 새로 연결 안 해도 됨)
  keepalive_timeout  65;
  
  # 클라이언트가 업로드할 수 있는 최대 파일 크기 (100MB까지 허용)
  client_max_body_size 100M;
  
  # 백엔드 API 서버 등록 - "service-api"라는 이름으로 192.168.1.225:9811 서버 지정
  upstream service-api{
    server 192.168.1.225:9811;
  }
  
  # 웹 서버 설정 시작
  server {
    # 80번 포트(일반 HTTP)로 들어오는 요청 받기
    listen 80;
    listen [::]:80;  # IPv6도 지원
    
    # 문자 인코딩 UTF-8로 설정 (한글 등 다국어 지원)
    charset utf-8;
    
    # [주석 처리됨] HTTPS로 리다이렉트하는 설정 (현재 비활성화 상태)
    #location / {
    #  return 301 https://$host$request_uri;
    #}
    
    # "/assistant/" 경로로 들어오는 요청 처리 (프론트엔드 웹페이지)
    location /assistant/ {
      # 실제 파일 위치는 /usr/share/nginx/html/ 폴더
      alias /usr/share/nginx/html/;
      
      # 기본 파일은 index.html
      index index.html;
      
      # 파일 찾는 순서: 1) 요청한 파일, 2) 요청한 폴더, 3) index.html
      # (React, Vue 같은 SPA에서 새로고침해도 작동하게 하는 설정)
      try_files $uri $uri/ /assistant/index.html;
    }   
    
    # "/assistant-api/" 경로로 들어오는 요청 처리 (백엔드 API)
    location /assistant-api/ {
      # 요청을 service-api(192.168.1.225:9811)로 전달 (프록시 역할)
      proxy_pass  http://service-api;
      
      # 백엔드 응답 대기 시간 90초
      proxy_read_timeout 90;
      
      # 백엔드 서버에 전달할 헤더 정보들 (원래 요청자 정보 유지)
      proxy_set_header Host $host;                      # 원래 호스트명
      proxy_set_header X-Real-IP $remote_addr;          # 실제 접속자 IP
      proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;  # 프록시 거친 IP 목록
      proxy_set_header X-Forwarded-Proto $scheme;       # 원래 프로토콜(http/https)
    }
    
    # [주석 처리됨] 컨텐츠 파일 제공 설정 (현재 비활성화 상태)
    #location /contents/ {
    # add_header Access-Control-Allow-Origin "*";  # CORS 허용
    # alias /sdata/alda_podcast_contents/;
    #}
    
    # 404 에러(페이지 없음) 발생시 index.html로 보내기
    error_page   404  /index.html;
    
    # 서버 에러(500번대) 발생시 50x.html 보여주기
    error_page   500 502 503 504  /50x.html;
    
    # 50x.html 파일 위치 지정
    location = /50x.html {
      root   /usr/share/nginx/html;
    }
  }
}
